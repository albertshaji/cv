%----------------------------------------------------------------------------------------
%   NEWCOMMANDS
%----------------------------------------------------------------------------------------

% New command for adding time to the margin
\newcommand{\when}[1]{\marginnote{\small{#1}}}

% New command for adding a new entry
\newcommand{\new}[4]{
    \medbreak
    \vspace{3mm}
    \marginnote{\small #1}
    \nopagebreak
    \textbf{#2}\\
    \nopagebreak
    \textit{#3}, #4
    \medskip
    }

% New command for section heading
\newcommand{\cat}[1]{
    \bigbreak
    \vspace{5mm}
    \pagebreak[1]
    {\Large\bfseries #1}
    \nopagebreak
    }

% New command for skill entry
\newcommand{\sky}[2]{
    \vspace{1mm}
    \marginnote{\small\bfseries #1}
    #2
    }
